1. EXECUTIVE SUMMARY

This document describes an architecture for implementing "Inner Voice" agents within LLM conversations. These are secondary agents that operate within conversational boundaries, reflecting on information preceding their activation, with their reflections fed back into the conversation context in real-time.

Core Concept: The conversation becomes a reflexive domain where the system observes itself through inner voice agents, creating eigenforms of understanding through iterative reflection.

Key Benefits

  • Improved Reasoning: Inner voices catch errors, biases, and gaps before final output
  • Transparency: Reflection traces show the reasoning process
  • Safety: Critical voices can flag harmful or problematic content
  • Depth: Multiple perspectives enrich the conversation
  • Self-Correction: System can revise before committing to output

5.1 Trigger Types

Always-On

// Inner voice activates on every turn
trigger: {
  type: "always"
}

Conditional

// Inner voice activates based on conditions
trigger: {
  type: "conditional",
  conditions: [
    {field: "confidence", operator: "lt", value: 0.7},
    {field: "topic_sensitivity", operator: "gt", value: 0.5},
    {field: "user_sentiment", operator: "eq", value: "negative"}
  ],
  logic: "ANY"  // or "ALL"
}

Sampling

// Inner voice activates probabilistically
trigger: {
  type: "sampling",
  probability: 0.3  // 30% of turns
}

Event-Driven

// Inner voice activates on specific events
trigger: {
  type: "event",
  events: [
    "topic_change",
    "contradiction_detected",
    "high_stakes_decision",
    "user_expresses_confusion"
  ]
}

Reflection Depth

// Inner voice activates based on reflection iterations
trigger: {
  type: "depth",
  min_depth: 2,  // After 2 reflection cycles
  max_depth: 5   // Stop after 5 cycles
}

5.2 Trigger Evaluation Engine

type TriggerEvaluator struct {
    context    *ConversationContext
    voices     []InnerVoice
}

func (e *TriggerEvaluator) ShouldActivate(voice InnerVoice) bool {
    switch voice.Trigger.Type {
    case "always":
        return true
    
    case "conditional":
        return e.evaluateConditions(voice.Trigger.Conditions)
    
    case "sampling":
        return rand.Float64() < voice.Trigger.Probability
    
    case "event":
        return e.hasEventOccurred(voice.Trigger.Events)
    
    case "depth":
        depth := e.context.ReflectionDepth()
        return depth >= voice.Trigger.MinDepth && 
               depth < voice.Trigger.MaxDepth
    
    default:
        return false
    }
}

    }
}

8. IMPLEMENTATION

8.1 Go Implementation

package innervoice

type InnerVoiceManager struct {
    voices       []InnerVoice
    evaluator    *TriggerEvaluator
    integrator   *ReflectionIntegrator
    boundaries   *BoundaryManager
}

type InnerVoice struct {
    ID           string
    Type         string
    SystemPrompt string
    Trigger      TriggerConfig
    Priority     string
    Weight       float64
    Model        string
}

type Reflection struct {
    VoiceID   string
    Content   string
    Timestamp time.Time
    Confidence float64
    Flags     []string  // e.g., "harmful", "uncertain", "verified"
}

func (m *InnerVoiceManager) ProcessTurn(ctx *ConversationContext, 
                                         userInput string) 
                                        (string, []Reflection, error) {
    
    // Step 1: Generate primary response
    outerOutput, err := m.generateOuterResponse(ctx, userInput)
    if err != nil {
        return "", nil, err
    }
    
    // Step 2: Activate inner voices
    reflections := []Reflection{}
    for _, voice := range m.voices {
        if m.evaluator.ShouldActivate(voice) {
            reflection, err := m.invokeVoice(voice, ctx, outerOutput)
            if err == nil {
                reflections = append(reflections, reflection)
            }
        }
    }
    
    // Step 3: Check for critical flags
    for _, r := range reflections {
        if contains(r.Flags, "harmful") {
            return "", reflections, ErrHarmfulContent
        }
    }
    
    // Step 4: Integrate reflections
    finalOutput := m.integrator.Integrate(outerOutput, reflections)
    
    // Step 5: Enforce boundaries
    if !m.boundaries.Validate(finalOutput) {
        return "", reflections, ErrBoundaryViolation
    }
    
    return finalOutput, reflections, nil
}

func (m *InnerVoiceManager) invokeVoice(voice InnerVoice, 
                                        ctx *ConversationContext,
                                        outerOutput string) 
                                       (Reflection, error) {
    
    prompt := fmt.Sprintf(`%s

Conversation Context:
%s

Primary Response to Reflect On:
%s

Provide your reflection:`, 
        voice.SystemPrompt, 
        ctx.FormatHistory(), 
        outerOutput)
    
    // Call LLM for this inner voice
    response, err := callLLM(voice.Model, prompt, voice.MaxTokens)
    if err != nil {
        return Reflection{}, err
    }
    
    return Reflection{
        VoiceID:    voice.ID,
        Content:    response,
        Timestamp:  time.Now(),
        Confidence: extractConfidence(response),
        Flags:      extractFlags(response),
    }, nil
}

8.2 Python Implementation

from dataclasses import dataclass
from typing import List, Optional
from enum import Enum

class VoicePriority(Enum):
    CRITICAL = "critical"
    HIGH = "high"
    MEDIUM = "medium"
    LOW = "low"

@dataclass
class InnerVoice:
    id: str
    voice_type: str
    system_prompt: str
    trigger: dict
    priority: VoicePriority
    weight: float
    model: str

@dataclass
class Reflection:
    voice_id: str
    content: str
    timestamp: datetime
    confidence: float
    flags: List[str]

class InnerVoiceOrchestrator:
    def __init__(self, voices: List[InnerVoice]):
        self.voices = voices
        self.context_buffer = ContextBuffer()
        self.integration_strategy = "pre-response"
    
    async def process_turn(self, user_input: str) -> Tuple[str, List[Reflection]]:
        # Add user input to context
        self.context_buffer.add_user_message(user_input)
        
        # Generate primary response (draft)
        draft_response = await self.generate_primary_response()
        
        # Activate inner voices
        reflections = []
        for voice in self.voices:
            if self.should_activate(voice):
                reflection = await self.invoke_voice(voice, draft_response)
                reflections.append(reflection)
        
        # Check for critical flags
        for r in reflections:
            if "harmful" in r.flags:
                raise HarmfulContentError(r)
        
        # Integrate reflections
        final_response = self.integrate_reflections(
            draft_response, 
            reflections
        )
        
        # Add to context
        self.context_buffer.add_assistant_message(final_response)
        
        return final_response, reflections
    
    def should_activate(self, voice: InnerVoice) -> bool:
        trigger = voice.trigger
        if trigger["type"] == "always":
            return True
        elif trigger["type"] == "conditional":
            return self.evaluate_conditions(trigger["conditions"])
        elif trigger["type"] == "sampling":
            return random.random() < trigger["probability"]
        return False

10.2 Safety Architecture

type SafetyLayer struct {
    contentFilter    *ContentFilter
    ethicsVoice      *InnerVoice
    boundaryManager  *BoundaryManager
    auditLogger      *AuditLogger
}

func (s *SafetyLayer) Validate(output string, 
                                 reflections []Reflection) 
                                (bool, error) {
    
    // Layer 1: Content Filter
    if !s.contentFilter.IsSafe(output) {
        s.auditLogger.Log("content_blocked", output)
        return false, ErrUnsafeContent
    }
    
    // Layer 2: Ethics Voice Check
    for _, r := range reflections {
        if r.VoiceID == "ethics-01" && 
           contains(r.Flags, "harmful") {
            s.auditLogger.Log("ethics_flag", r.Content)
            return false, ErrEthicsViolation
        }
    }
    
    // Layer 3: Boundary Validation
    if !s.boundaryManager.Validate(output) {
        return false, ErrBoundaryViolation
    }
    
    // Layer 4: Audit Log
    s.auditLogger.Log("output_approved", map[string]interface{}{
        "output_length": len(output),
        "reflections_count": len(reflections),
        "timestamp": time.Now(),
    })
    
    return true, nil
}

// Parallel Execution Example
func (m *InnerVoiceManager) InvokeAllVoices(ctx Context, 
                                            voices []InnerVoice) 
                                           []Reflection {
    results := make(chan Reflection, len(voices))
    wg := sync.WaitGroup{}
    
    for _, voice := range voices {
        wg.Add(1)
        go func(v InnerVoice) {
            defer wg.Done()
            
            // Timeout for each voice
            done := make(chan Reflection, 1)
            go func() {
                r, _ := m.invokeVoice(v, ctx, "")
                done <- r
            }()
            
            select {
            case r := <-done:
                results <- r
            case <-time.After(v.Timeout):
                // Voice timed out, skip
            }
        }(voice)
    }
    
    wg.Wait()
    close(results)
    
    var reflections []Reflection
    for r := range results {
        reflections = append(reflections, r)
    }
    
    return reflections
}

12.3 Reflection Quality Score

// Measure if reflections are actually useful
func CalculateReflectionQuality(reflections []Reflection, 
                                original string, 
                                revised string) float64 {
    
    score := 0.0
    
    for _, r := range reflections {
        // Did this reflection change the output?
        if influencedOutput(r, original, revised) {
            score += 0.3
        }
        
        // Was the reflection accurate?
        if reflectionWasCorrect(r, revised) {
            score += 0.3
        }
        
        // Was it actionable?
        if reflectionWasActionable(r) {
            score += 0.2
        }
        
        // Was it concise?
        if len(r.Content) < 500 {
            score += 0.2
        }
    }
    
    return score / float64(len(reflections))
}
